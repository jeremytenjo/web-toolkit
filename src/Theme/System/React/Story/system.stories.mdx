## Usage

Pass a `styles` object with styles, arrays generate media queries set in the styles components theme provider

if mediaQueries = [700, 1200]  color: ['primary', 'secondary'] === 0-700 = primary, 700-1200=secondary

```js
import System from '@tenjojeremy/web-toolkit/build/Theme/System/React/system.index'
 ```

```js
import React from 'react'
import styled from 'styled-components'

import System from '@tenjojeremy/web-toolkit/build/Theme/System/React/system.index'

const Wrapper = styled.div`
  ${system}
`

export default () => (
  <Wrapper styles={{ color: ['primary', 'secondary'], margin:'xl' }}>
    Theme system
  </Wrapper>
)

 ```
